#"Join Metas" =

Table.NestedJoin(

#"Filtra Submarcas",

{

"MES_REFERENCIA",

"VENDEDOR",

"SUBMARCA"

},

Fato_Metas,

{

"MES_REFERENCIA",

"VENDEDOR",

"SUBMARCA"

},

"Meta",

JoinKind.LeftOuter

),





// ============================================================

// Expande o valor da meta para análise comparativa

// ============================================================



#"Expand Meta" =

Table.ExpandTableColumn(

#"Join Metas",

"Meta",

{

"META_FATURAMENTO"

},

{

"META_FATURAMENTO"

}

),





// ============================================================

// Cálculo do percentual de atingimento

//

// Fórmula:

//

// Faturamento Realizado / Meta

//

// Exemplo:

// Venda: 120.000

// Meta: 100.000

//

// Resultado: 120%

// ============================================================



#"Atingimento" =

Table.AddColumn(

#"Expand Meta",

"Atingimento_Percentual",

each

if [META_FATURAMENTO] = null

or [META_FATURAMENTO] = 0

then null

else

[Faturamento_Realizado] / [META_FATURAMENTO],

type number

),





// ============================================================

// Classificação do atingimento da meta

//

// Regra:

// >= 100% = Meta atingida

// < 100% = Meta não atingida

// ============================================================



#"Meta Atingida" =

Table.AddColumn(

#"Atingimento",

"Meta_Atingida",

each

if [Atingimento_Percentual] <> null

and [Atingimento_Percentual] >= 1

then "Sim"

else "Não",

type text

),





// ============================================================

// Cálculo de pontuação do vendedor

//

// Somente recebe pontos quando atingir a meta.

//

// Regras da campanha:

//

// Energéticos = 200 pontos

// Isotônicos = 100 pontos

// Cerveja Premium = 150 pontos

// ============================================================



#"Pontos Vendedor" =

Table.AddColumn(

#"Meta Atingida",

"Pontos_Vendedor",

each

if [Meta_Atingida] = "Sim"

and [SUBMARCA] = "ENERGÉTICOS"

then 200



else if [Meta_Atingida] = "Sim"

and [SUBMARCA] = "ISOTÔNICOS"

then 100



else if [Meta_Atingida] = "Sim"

and [SUBMARCA] = "CERVEJA PREMIUM"

then 150



else 0,

Int64.Type

),





// ============================================================

// Cálculo de pontuação do supervisor

//

// Regras da campanha:

//

// Energéticos = 40 pontos

// Isotônicos = 50 pontos

// Cerveja Premium = 30 pontos

// ============================================================



#"Pontos Supervisor" =

Table.AddColumn(

#"Pontos Vendedor",

"Pontos_Supervisor",

each

if [Meta_Atingida] = "Sim"

and [SUBMARCA] = "ENERGÉTICOS"

then 40



else if [Meta_Atingida] = "Sim"

and [SUBMARCA] = "ISOTÔNICOS"

then 50



else if [Meta_Atingida] = "Sim"

and [SUBMARCA] = "CERVEJA PREMIUM"

then 30



else 0,

Int64.Type

),





// ============================================================

// Formatação final

// Converte o atingimento para percentual para apresentação

// no Power BI.

// ============================================================



#"Tipo Alterado" =

Table.TransformColumnTypes(

#"Pontos Supervisor",

{

{

"Atingimento_Percentual",

Percentage.Type

}

}

)





in



#"Tipo Alterado"
Dim_PDVlet

Fonte = Base_SELL_OUT,

#"Outras Colunas Removidas" = Table.SelectColumns(Fonte,{"PDV_LOJA"}),

#"Duplicatas Removidas" = Table.Distinct(#"Outras Colunas Removidas")

in

#"Duplicatas Removidas"
Medidaslet

Origem = Table.FromRows(Json.Document(Binary.Decompress(Binary.FromText("i44FAA==", BinaryEncoding.Base64), Compression.Deflate)), let _t = ((type nullable text) meta [Serialized.Text = true]) in type table [#"Coluna 1" = _t]),

#"Colunas Removidas" = Table.RemoveColumns(Origem,{"Coluna 1"})

in

#"Colunas Removidas"
Parâmetro{

("DISTRIBUIDOR", NAMEOF('Fato_Vendas'[DISTRIBUIDOR]), 0),

("SUPERVISOR", NAMEOF('Fato_Vendas'[SUPERVISOR]), 1),

("VENDEDOR", NAMEOF('Fato_Vendas'[VENDEDOR]), 2)

}
Categoria{

("VENDEDOR", NAMEOF('Dim_Vendedor'[VENDEDOR]), 0),

("SUPERVISOR", NAMEOF('Dim_Supervisor'[SUPERVISOR]), 1),

("SUBMARCA", NAMEOF('Dim_Produto/Marca'[SUBMARCA]), 2),

("DISTRIBUIDOR", NAMEOF('Dim_Distribuidor'[DISTRIBUIDOR]), 3)

}
rankink{

("VENDEDOR", NAMEOF('Fato_Resultado_Campanha'[VENDEDOR]), 0),

("SUPERVISOR", NAMEOF('Fato_Resultado_Campanha'[SUPERVISOR]), 1)

}
Parametro Comercial{

("SUPERVISOR", NAMEOF('Fato_Resultado_Campanha'[SUPERVISOR]), 0),

("VENDEDOR", NAMEOF('Fato_Resultado_Campanha'[VENDEDOR]), 1)

}
Parâmetro 2{

("Período", NAMEOF('Dim_Calendario'[NomeMês]), 0),

("SUBMARCA", NAMEOF('Fato_Resultado_Campanha'[SUBMARCA]), 1)

}
